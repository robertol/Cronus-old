OnAgitBreak:

	if (.@GID <= 0) {
		set .@notice$,"O Jogador "+strcharinfo(0)+" ("+getcharid(0)+") Quebrou o Emperium do Castelo: "+strnpcinfo(2)+" enquanto estava sem Guilda. As informações não Serão Salvas e o Emperium Reaparecerá.";
 		logmes .@notice$; debugmes .@notice$;

		end;

	if (.@Economy < 0) set .@Economy, 0;

	// Adjust Defense Invest Level for Castle
	set .@Defence,GetCastleData(strnpcinfo(2),3) - 5;
	if (.@Defence < 0) set .@Defence, 0;
	SetCastleData strnpcinfo(2), 3, .@Defence;

	SetCastleData strnpcinfo(2),1, .@GID;

	mapannounce strnpcinfo(2),"O Emperium foi Quebrado!",bc_map|bc_woe,"0x00CCFF",FW_NORMAL,12;
	MapRespawnGuildID strnpcinfo(2),.@GID,6;

	donpcevent strnpcinfo(0)+"::OnRecvCastle";
	disablenpc "Kafra Staff#"+strnpcinfo(2);

		SetCastleData strnpcinfo(2), .@i, 0;

	end;

OnAgitEnd:
	if (strnpcinfo(0) == "Gld_Agit_Manager") end;

	if (GetCastleData(strnpcinfo(2),1)) {
		KillMonster strnpcinfo(2),"Agit#"+strnpcinfo(2)+"::OnAgitBreak";
	}
	end;

// Occupying Guild has been disbanded.

	disablenpc "Kafra Staff#"+strnpcinfo(2);

	// Wait before refreshing guild information.

	Announce "O Castelo [" + GetCastleName(strnpcinfo(2)) + "] foi abandonado.",0;
	donpcevent strnpcinfo(0)+"::OnRecvCastle";
	end;

-	script	Kafra Staff#woe::guildkafra	-1,{

	set .@GID,GetCastleData(strnpcinfo(2),1);

	if (compare(strnpcinfo(2),"aldeg")) {

	else if (compare(strnpcinfo(2),"gefg")) {
		setarray .@destination$[0],"Geffen","geffen";

	}
	else if (compare(strnpcinfo(2),"payg")) {
		setarray .@destination$[0],"Payon","payon";
		setarray .@coordinates[0],70,100;
	}

		setarray .@destination$[0],"Prontera","prontera";
		setarray .@coordinates[0],278,211;

	else {
		end;
	}

	cutin "kafra_01",2;
	if (getcharid(2) == .@GID) {
		mes "[Funcionária Kafra]";
		mes "Bem Vindo. Membro da Guilda ^ff0000" + GetGuildName(.@GID) + "^000000.";
		mes "A Corporação Kafra estará sempre ao seu lado.";
		next;
		switch (select("Usar Armazém:Usar Serviço de Teletransporte:Alugar Carrinho:Cancelar")) {
		case 1:
			if (getskilllv("NV_BASIC") < 6) {
				mes "[Funcionária Kafra]";
				mes "Me Desculpe, mas Voce";
				mes "precisa ter as";
				mes "habilidades básicas nível 6 para";
				mes "usar o Armazém.";
				close2;
				cutin "",255;
				end;
			}

			mes "seu armazém para Voce.";

			mes "os Serviços Kafra.";

			cutin "",255;

			mes "[Funcionária Kafra]";
			mes "Por Favor escolha";
			mes "seu Destino.";
			next;
			switch (select(.@destination$+" -> 200z:Cancel")) {
			case 1:
				if (Zeny < 200) {
					mes "[Funcionária Kafra]";
					mes "Me desculpe, mais voce não tem";

				set RESRVPTS,RESRVPTS+2;

				end;
			case 2:
				close2;
				cutin "",255;
				end;
			}
		case 3:
			if (BaseClass != Job_Merchant) {
				mes "[Funcionária Kafra]";
				mes "Me Desculpe, mas o serviço";
				mes "de alguem de Carrinho";
				mes "está disponível apenas para Mercadores,";
