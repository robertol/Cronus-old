// Nota: "Comentários" são os textos com duas barras "//" no início da linha
// Qualquer texto que for comentado não será considerado pelo servidor, e serve
// apenas como informação/referência.

// O login-server reconhece a interface com este endereço de IP.
// NOTA: Isto permite rodar vários servidores em múltiplas interfaces
// usando a mesma porta para cada servidor.
//bind_ip: 127.0.0.1
 
// Porta do Login-Server
login_port: 6900

//Formato da data e hora que será exibido antes de todas as mensagens.
//Pode ter no máximo 20 caracteres.
//Formatos comuns:
// %I:%M:%S %p (hora:minuto:segundo 12 horas, formato AM/PM)
// %H:%M:%S (hora:minuto:segundo, formato 24 horas)
// %d/%b/%Y (dia/mês/ano)
//Para informações detalhadas, consulte o manual do comando 'strftime()'.
//timestamp_format: [%d/%b %H:%M]

//A saída redirecionada contém sequências de escape (códigos de cores)
stdout_with_ansisequence: no

//Torne a saída do servidor mais silenciosa omitindo certos tipos de mensagem:
//1: Esconder mensagens de Informação
//2: Esconder mensagens de Status
//4: Esconder Registro
//8: Esconder mensagens de Alerta
//16: Esconder Erros e mensagens de Erro SQL.
//32: Esconder Debugs
//Exemplo: "console_silent: 7" esconde informações, status e alertas (1+2+4)
console_silent: 0

// Comandos do Console
// Permitir uso de comandos (on/off)
// Isto previne o uso do >& log.file
console: off

// Permitir criação de contas usando _M/_F?
new_account: yes

// Você deseja que as criações de contas sejam de no mínimo 4 caracteres para nome de usuário/senha?
// Escolha "yes", ao menos que você use em seu cliente a opção: "Disable 4 LetterUserID/Password".
new_acc_length_limit: yes

// Proteção contra criação de contas em tempo abusivo
// allowed_regs é o número de registros permitidos em time_allowed (em segundos)
allowed_regs: 1
time_allowed: 10

// Nome do arquivo de Log. Todas as operações recebidas pelo servidor são guardadas neste arquivo.
login_log_filename: log/login.log

// Gravar o login-server?
// NOTA: O login-sql precisa os logs do login para habilitar banimetos por erros de senha dinâmicos.
log_login: yes

// Indique como exibir a data/hora nos logs, para jogadores, etc.
date_format: %d-%m-%y %H:%M:%S

// Group ID Necessário para Conectar-se ao Servidor
// -1: Desativado
// 0 ou Maior: group id
group_id_to_connect: -1

// Group ID  Minino Requerido para se Conectar so Servidor.
// Se a Configuração group_id_to_connect estiver habilitada essa opção nao Funcionará.
// -1: desativado
// 0 ou maior: group id
min_group_id_to_connect: -1

// Iniciar segundos adicionais para limite de tempo de criação de conta
// -1: novas contas são criadas com tempo ilimitado (valor padrão)
// 0 ou mais: novas contas foram criadas pela adição de tempo (em segundos) para
// o tempo atual (escolher limite de tempo)
start_limited_time: -1

// Checar a versão do client escolhida no clientinfo ?
check_client_version: no

// Que versão deve ser escolhida para que a conexão seja aceita? (se a opção acima for habilitada...)
client_version_to_connect: 20

// Guardar senhas com hashes MD5 ao invés de campos de texto ?
// OBS.: Não irá funcionar com clients que usam <passwordencrypt>
// OBS.: Se habilitar essa função, lembre que as senhas do banco de dados
//   devem estar em formato MD5 também (incluindo as de servidor)!
use_MD5_passwords: no

// Características do Ipban (somente SQL)
ipban.enable: yes
//ipban.sql.db_hostname: 127.0.0.1
//ipban.sql.db_port: 3306
//ipban.sql.db_username: ragnarok
//ipban.sql.db_password: ragnarok
//ipban.sql.db_database: ragnarok
//ipban.sql.codepage: 
//ipban.sql.ipban_table: ipbanlist
// Dynamic password failure ipban system
ipban.dynamic_pass_failure_ban: yes
ipban.dynamic_pass_failure_ban_interval: 5
ipban.dynamic_pass_failure_ban_limit: 7
ipban.dynamic_pass_failure_ban_duration: 5

// Intervalo (em segundos) para varrer bans expirados de IP. 0 = desabilitado. padrão = 60.
// NOTA: Mesmo que desabilitado, os bans expirados de IP serão varridos ao ligar/desligar o login-server.
// Jogadores irão poder logar se a entrada de ipban existir e o tempo de expiração já tiver passado.
ipban_cleanup_interval: 60

// Intervalo (em minutos) para executar atualização de DNS/IP. Desabilitado por padrão.
// Habilite caso seu servidor usa um IP dinâmico que muda com o tempo.
//ip_sync_interval: 10

// Lista Negra de DNS
// Caso habilitado, todas as conexões serão comparadas as da lista
// nos específicos dnsbl_servers (separe os dsnbl_servers por vírgulas)
use_dnsbl: no
dnsbl_servers: bl.blocklist.de, socks.dnsbl.sorbs.net
// Here are some free DNS Blacklist Services: http://en.wikipedia.org/wiki/Comparison_of_DNS_blacklists
//==============================================================================
//   servidores_dnsbl                   Descrição
// - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
// bl.blocklist.de                 IP que atacar outros servidores/honeypots sobre SSH, FTP, IMAP, etc.
// ircbl.ahbl.org                  AHBL (proxies abertos, computadores comprometidos, os spammers de comentário)
// safe.dnsbl.sorbs.net            Todas as zonas em dnsbl.sorbs.net exceto "recente" e "escalações"
// sbl-xbl.spamhaus.org            Spamhaus lista negra (spammers, proxies abertos)
// socks.dnsbl.sorbs.net           Servidores de proxy SOCKS abertos
// tor.ahbl.org                    Current tor relay and exit nodes

// Qual mecanismo da conta usar.
// 'auto' seleciona o primeiro mecanismo (sql)
// (o padrão é auto)
account.engine: auto

// Configuração dos dados do armazenamento de contas
// SQL
//account.sql.db_hostname: 127.0.0.1
//account.sql.db_port: 3306
//account.sql.db_username: ragnarok
//account.sql.db_password: ragnarok
//account.sql.db_database: ragnarok
//account.sql.codepage: 
//account.sql.case_sensitive: no
//account.sql.account_db: login
//account.sql.accreg_db: global_reg_value

// Verificação de hash MD5 do cliente
// Se ativado, o login-server irá verificar se coincide com o hash do cliente
// o valor abaixo não irá se conectar a clientes adulterados.
// NOTA: consulte doc\md5_hashcheck.txt para obter mais detalhes.
client_hash_check: off

// Ponha seus hashes de clientes aqui. Um jogador pode entrar no server
// usando um hash de um group_id igual ou menor que a conta tem.
// Formato: group_id, hash
// Exemplos:
client_hash: 0, 113e195e6c051bb1cfb12a644bb084c5
//client_hash: 99, cb1ea78023d337c38e8ba5124e2338ae
