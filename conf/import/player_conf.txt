//--------------------------------------------------------------
// Nota 1: O valor é configurado por (on/off, yes/no ou 1/0)
// Nota 2: O valor é em porcentagem (100 é o mesmo que 100%)
//--------------------------------------------------------------

// Taxa máxima de HP para jogador. (O padrão é 100)
hp_rate: 100

// Taxa máxima de SP para jogador. (O padrão é 100)
sp_rate: 100

// Cartas e atributos para a mão esquerda também são aplicados na mão direita? (Nota 1)
// (Definido para 'yes' em servidores oficiais)
left_cardfix_to_right: yes

// Quantidade de HP que um jogador irá reviver, 0 é o padrão.
// (A unidade é a porcentagem total de HP, 100 é o HP por completo, 0 significa 1 de HP.)
restart_hp_rate: 70

// Quantidade de SP que um jogador irá reviver, 0 é o padrão.
// (A unidade é a porcentagem total de SP, 100 é o SP por completo, 0 significa 1 de SP.)
restart_sp_rate: 70

// Um jogador comum pode burlar a árvore de habilidades? (Nota 1)
player_skillfree: no

// Quando definido para yes, força pontos de habilidade ganhos na 1ª classe serem distribuídos nela
// e força habilidades de aprendiz serem distribuídas nas Habilidades Básicas. (Nota 1)
player_skillup_limit: yes

// Habilidades de quest podem ser aprendidas? (Nota 1)
// Configurando para yes poderá abrir exploits em seu servidor!
quest_skill_learn: no

// Quando as habilidades são resetados, as habilidades de quest também são? (Nota 1)
// Configurando para yes poderá abrir exploits em seu servidor!
// NOTA: Se você tiver quest_skill_learn definida para yes, habilidades de quests sempre serão resetadas.
quest_skill_reset: no

// É obrigatório possuir Habilidades Básicas para sentar, realizar negociações, criar grupos ou salas de chat? (Nota 1)
basic_skill_check: yes

// Quando movido, ou mandado para um mapa, quanto tempo antes do monstro detectar a presença do jogador sem que ele se mova (tempo em milisegundos)
// Se você atacar um monstro, ele irá atacá-lo independente desta configuração.
player_invincible_time: 5000

// Intervalo de recuperação natural de HP. (em milisegundos)
natural_healhp_interval: 6000

// Intervalo de recuperação natural de SP. (em milisegundos)
natural_healsp_interval: 8000

// Intervalo de recuperação automático de habilidades de cura. (em milisegundos)
natural_heal_skill_interval: 10000

// O peso máximo que um personagem pode carregar para que a recuperação de hp natural seja interrompida. (em %)
natural_heal_weight_rate: 50

// Máxima velocidade de ataque. (O padrão é 190, o MÁXIMO é 199)
max_aspd: 190

// Máxima taxa de velocidade de movimento (200 é o dobro da velocidade normal)
max_walk_speed: 300

// O mesmo que max_aspd, mas para as 3ª classes. (Padrão 193, Máximo permitido 199)
max_third_aspd: 193

// Máxima taxa de velocidade de movimento (200 é o dobro da velocidade normal)
max_walk_speed: 300

// HP máximo. (O padrão é 1000000)
max_hp: 1000000

// SP máximo. (O padrão é 1000000)
max_sp: 1000000

// Limite máximo de stats. (agi, str, etc.)
max_parameter: 300

// O mesmo que max_parameter, mas para as 3ª classes.
max_third_parameter: 300

// Mesmo do max_parameter, mas para classe bebê.
max_baby_parameter: 300

// // Mesmo do max_parameter, mas para as 3ª classes bebê.
max_baby_third_parameter: 300

// Máximo de def/mdef
// NOTA: não afeta habilidades e efeitos como Corpo Fechado
// Se weapon_defense_type não for zero, não será aplicada a máx def.
// Se magic_defense_type não for zero, não será aplicada a máx mdef.
max_def: 99

// Conversão de bônus de Def para Def2. Se a def/mdef da armadura exceder a max_def,
// o restante é convertido em vit def/int mdef usando este multiplicador
// (ex: se definido para 10, cada ponto restante gerará 10 pontos de vit def)
over_def_bonus: 0

// O peso máximo que um carrinho pode levar.
max_cart_weight: 8000

// Previnir que um jogador saia do jogo enquanto esteja sendo atacado (em ms, 0 desativa)?
prevent_logout: 10000

// Exibir hp/sp "drenados" de ataques normais? (Ie: Carta Mosca Caçadora)
show_hp_sp_drain: no

// Exibir hp/sp ganhos ao matar monstros? (Ie: Carta Deletério Voador)
show_hp_sp_gain: yes

// Caso definido, quando A aceita B como amigo, B será adicionado a lista de amigos de  A,
// do contrário, somente A aparecerá na lista de amigos de B.
// NOTA: esta configuração não habilita a remoção mútua de amigos, somente a adição.
friend_auto_add: yes

// Convite de grupo/clã ou negociações simultâneas são automaticamente rejeitadas?
invite_request_check: yes

// Jogadores deixarão sua "Caveira" quando morrerem?
// 0 = Desabiltiado
// 1 = Apenas em mapas PvPs
// 2 = Em todas as situações
bone_drop: 2

// Personagens montados (em Peco) terão seus tamanhos aumentados
// 0 = não
// 1 = somente Classes Normais em Peco terão tamanho grande
// 2 = somente Classes Bebês em Peco terão tamanho médio
// 3 = As Classes Normais em Peco terão tamanho grande
//	e as Classes Bebês em Peco terão tamanho médio
character_size: 0

// Personagens inativos recebem itens do autoloot?
// Defina o tempo em segundos onde um personagem inativo não receberá
// itens do Autoloot (0: desativado).
idle_no_autoloot: 600

// Distância minima a que uma loja deve ser colocada de um npc.
// Padrão: 3 (0: desativado).
min_npc_vendchat_distance: 3
