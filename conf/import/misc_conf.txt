// Nota 1: O valor é configurado por (on/off, yes/no ou 1/0)
// Nota 2: O valor é em porcentagem (100 é o mesmo que 100%)
// Nota 3: O valor é em bit. Caso não haja descrição,
//         assuma como valores (1: Pc, 2: Mob, 4: Pet, 8: Homun)

// Modo PK. O servidor inteiro fica em PvP (exceto as cidades). A perda de experiência é dobrada quando se morre por um jogador.
// Quando matam monstros 20 níveis acima do seu, recebem 15% de exp adicional e 25% de mais itens. 
// Existe o nopvp.txt para configurar mapas que não possuem pk neste modo. Aprendizes não podem ser atacados ou atacar.
// O contador de pvp e o rank também serão desabilitados.
// Nota: Se pk_mode for definida para 2 ao invés de 1 (yes), os jogadores receberão uma
//   penalidade de manner de 5 toda vez que matar outro jogador (veja a configuração de manner_system
//   para configurar como isto afetará os jogadores)
pk_mode: 1

// Configuração do Manner/karma. Especifica como o manner irá
// afetar os jogadores (some quando necessário):
//  0: Sem penalidades.
//  1: Desabilita o chat (incluí whispers, msgs de grupo/clã, etc)
//  2: Desabilita o uso de habilidades
//  4: Desabilita o uso de comandos
//  8: Desabilita usar/pegar/dropar itens
// 16: Desabilita criação de salas de chats e vendas
manner_system: 0

// Para servidores PK. Modifique isto para o nível mínimo que os jogadores podem ficar em PK
pk_min_level: 1

// Para servidores PK. Especifica o nível máximo de diferença entre
// jogadores para poderem se atacar. 0 desabilita o limite.
pk_level_range: 0

// Para servidores PK. Configurações de dano, que seguem a mesma lógica
// das configurações da GdE (veja guild.conf)  
pk_short_attack_damage_rate: 100
pk_long_attack_damage_rate: 100
pk_weapon_attack_damage_rate: 100
pk_magic_attack_damage_rate: 100
pk_misc_attack_damage_rate: 100

// Exibir uso de habilidades no console? (somente debug) (Padrão: off) (Nota 3)
skill_log: off

// Exibir log da battle? (somente debug) (Padrão: off) (Nota 1)
battle_log: off

// Exibir outros logs? (somente debug) (Padrão: off) (Nota 1)
etc_log: off

// Debugar pontos de warp? Se sim, pontos de warp irão aparecer como flags.(Nota 1)
// Isto também irá rodar no start do servidor uma checagem de warps para exibir quais warps estão diretamente na
// parte superior de npcs com ontouch (que pode causar warps com loopbacks infinitos)
warp_point_debug: no

// O servidor será iniciado à noite (yes) ou dia (no) (Nota 1)
night_at_start: yes

// Defina a duração do dia em milisegundos (padrão: 1800000 = 30 min)
// Escolha 0 para desabilitar o ciclo do dia (mas não o comando @day).
// Excetuando 0, o mínimo é 60000 (1 minuto)
day_duration: 1800000

// Defina a duração da noite em milisegundos (padrão: 1800000 = 30 min)
// Escolha 0 para desabilitar o ciclo de noite (mas não o comando @night).
// Excetuando 0, o mínimo é 60000 (1 minuto)
night_duration: 1800000

// Habilitar duelos em mapas com pvp (Nota 1)
duel_allow_pvp: yes

// Habilitar duelos em mapas com gvg (Nota 1)
duel_allow_gvg: yes

// Permitir o uso de teleporte durante o duelo (Nota 1)
duel_allow_teleport: yes

// Sair automaticamente de um duelo quando morrer (Nota 1)
duel_autoleave_when_die: yes

// Intervalo em minutos para utilizar "@duel"
duel_time_interval: 1

// Restringir o duelo somente no mesmo mapa (Nota 1)
duel_only_on_same_map: yes

// Determine o número máximo de personagens que podem ocupar uma única célula.
// NOTA: Para esta opção ter efeito você deve compilar o servidor com o
// suporte ao valor configurado no "cell_stack_limit" (see src/map/map.h)
cell_stack_limit: 1

// Permitir usar autotrade em mapas somente com a flag para tal? (Nota 1)
// Defina para "no" e autotrade será habilitado em todo o servidor
// Defina para "yes" para habilitar o autotrade apenas em mapas com a mapflag "autotrade"
at_mapflag: no

// Configure isto para quantos minutos de autotrade serão necessários para tirar um personagem do servidor.
at_timeout: 720

// Taxa por hora no sistema de leilões. O padrão é 12000
auction_feeperhour: 12000

// Preço máximo de venda nos leilões
auction_maximumprice: 500000000

// Tempo mínimo entre o uso do search store em segundos.
searchstore_querydelay: 10

// Quantidade máxima de resultados do search store, antes de ser cancelado.
searchstore_maxresults: 30

// A perda ou o ganho de pontos de cash serão exibidos (Nota 1).
// Padrão: no
cashshop_show_points: yes

// O status do sistema de e-mails é exibido ao logar.
// Padrão: 0
// 0 = Não
// 1 = Sim
// 2 = Sim, quando houver e-mails não-lidos.
mail_show_status: 0
