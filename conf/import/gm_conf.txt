// A quantidade máxima de monstros que podem ser invocados por atcommand (0 significa quantidade ilimitada)
atcommand_spawn_quantity_limit: 30

// Número máximo de clones escravos que pode-se ter usando o comando @slaveclone. (0 significa quantidade ilimitada)
atcommand_slave_clone_limit: 5

// Se 'no', comandos exigirão o nome exato do jogador. Se 'yes', digitando parcialmente 
// o nome irá funcionar, quando houver apenas um resultado de todos os jogadores 
// no mapa atual.
partial_name_scan: yes

// (@) @allstats/@str/@agi/@vit/@int/@dex/@luk
// Permite aos gms ultrapassar o limite máximo dos status? ( se "yes" status de gm poderão chegar a 32k) Padrão: no
atcommand_max_stat_bypass: no

// Banir pessoas que tentam burlar a negociação.
// Duração do banimento, em minutos (padrão: 5). Para desabilitar, defina 0.
ban_hack_trade: 600000

// Necessita que RENEWAL_DROP esteja ativo (src/map/config/renewal.h)
// Modifica @mobinfo para informar ao usuário a chance de drop real pela formula do renewal_drop.
// Modifica @iteminfo para não exibir a chance mínima de drop do item (desde que possa indicar o nível do mob)
atcommand_mobinfo_type: 0
