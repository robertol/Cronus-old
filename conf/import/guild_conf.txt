// Quando um clã é criado, o Emperium é usado? (Nota 1)
guild_emperium_check: yes

// Limite da taxa de doação para um membro do clã.
guild_exp_limit: 50

// Máximo de castelos que um clã pode conquistar (0 = ilimitado)
guild_max_castles: 0

// Ativar delay de habilidades do clã por reconexão? (Nota 1)
// O ajuste oficial é "yes", caso contrário permitirá líderes de clãs reconectarem para cancelar o delay de 5 minutos.
guild_skill_relog_delay: yes

// Ajuse de danos de GdE contra guardiões (Nota 2)
castle_defense_rate: 100

// Ajuste de danos físicos (excluí habilidades) para GdE (Guils Vs Guild) (Nota 2)
gvg_short_attack_damage_rate: 80

// Ajuste de danos de alcance (excluí habilidades) para GdE (Guild Vs Guild) (Nota 2)
gvg_long_attack_damage_rate: 80

// Ajuste de danos de habilidades com uso de arma para GdE (Guild Vs Guild) (Nota 2)
gvg_weapon_attack_damage_rate: 60

// Ajuste de danos de habilidades com uso de magias para GdE (Guild Vs Guild) (Nota 2)
gvg_magic_attack_damage_rate: 60

// Ajuste de danos de outras habilidades para GdE (Guild Vs Guild) (Nota 2)
gvg_misc_attack_damage_rate: 60

// Penalidade na Esquisa em mapas GvG. O padrão oficial é 20 (Nota 2)
// NOTA: A configuração é feita em porcentagem (%), então 20 = -20% da esquiva total
gvg_flee_penalty: 20

// A habilidade 'Glória do Clã' pode ser aprendida,
// e a mudança de emblemas a exige? (Nota 1)
// P.S: Esta habilidade não é implementada em servidores oficiais.
require_glory_guild: no

// Limite de alianças de um clã. O valor vai de 0 a 3.
// Se você deseja modificar, execute uma limpeza na tabela relacionada a alianças de clãs.
// Padrão: 3
max_guild_alliance: 3
