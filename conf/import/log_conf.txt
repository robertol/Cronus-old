// Habilitar Registros? (Nota 3)
// 0x00000 - Não registrar nada
// 0x00001 - (T) Negociações
// 0x00002 - (V) Transações de vendas
// 0x00004 - (P) Itens jogados/pegos por jogadores
// 0x00008 - (L) Itens deixados/pegos por monstros
// 0x00010 - (S) Transações de NPCs (compra/venda)
// 0x00020 - (N) Transações de scripts (itens deletados/adquiridos por quests)
// 0x00040 - (D) Itens roubados de monstros (Furto/Mãos Leves)
// 0x00080 - (C) Itens usados por jogadores (Consumidos/pet&hom&merc Comida/Itens usados por Habilidades&ataques)
// 0x00100 - (O) Itens produzidos/ingredientes de jogadores
// 0x00200 - (U) Prêmios em forma de itens dados por MvPs
// 0x00400 - (A) Itens criados/apagados por jogadores (através de comandos @/#)
// 0x00800 - (R) Itens colocados/removidos do armazém.
// 0x01000 - (G) Itens colocados/removidos do armazém do clã.
// 0x02000 - (E) Transações realizadas pelo correio.
// 0x04000 - (I) Transações realizadas pelo leilão.
// 0x08000 - (B) Transações realizadas pela janela de compra.
// 0x10000 - (X) Todas as outras (rentals expiring/inserting cards/items removed by item_check/
//           rings deleted by divorce/pet egg (un)hatching/pet armor (un)equipping/Weapon Refine skill/Remove Trap skill)
// Examplo: Negociações+Transações de vendas+Transações de NPCs (compra/venda)+Itens produzidos/ingredientes de jogadores: 1+2+32+1024 = 1059
enable_logs: 0xFFFFF

// Utilizar Registros em MySQL? (Nota 1)
sql_logs: yes

// FILTROS DE LOG
//=============================================================
// Se alguma condição for verdadeira então o item será registrado no log.
// 0 = Nunca faz log.
// 1 = Faz log de qualquer coisa.
// Filtro avançado de bits: ||
// 0002 - Itens de cura. (0)
// 0004 - Outros itens. (items(3) + flechas(10))
// 0008 - Usavéis. (items(2) + lures, pergaminhos(11))
// 0016 - Armas. (4)
// 0032 - Escudos, armaduras, capacetes, acessorios, etc. (5)
// 0064 - Cartas. (6)
// 0128 - Acessorios para bichinhos de estimação. (8) + ovos (7) (monstros não dropam este tipo de itens mas usaremos o mesmo sistema para TODOS os logs)
// 0256 - Log de itens caros. (se o preço for >= price_items_log)
// 0512 - Log de grande quantidade de itens. (se a quantidade for maior ou igual ao valor da função "amount_items_log")
// 1024 - Log de itens refinados. (se a taxa de refinamento for maior ou igual ao valor da função "refine_items_log")
// 2048 - Log de itens raros. (se a chance de queda for maior ou igual ao valor da função "rare_items_log")

// Exemplos: (filtros de log)
// log_filter: 1  = Faz log de qualquer item.
// log_filter: 2	= Faz log apenas de itens de cura.
// log_filter: 4	= Faz log apenas de outros itens (ETC), itens de cura e usáveis.
// log_filter: 64	= Faz log apenas de cartas.
// log_filter: 322	= Faz log apenas de itens de cura, cartas e itens que o preço for >= price_items_log.
// log_filter: 4080	= Faz log de todos os itens (incluindo raros, grande quantidade) exceto itens de cura, ETC, flechas e usáveis.
log_filter: 1

// Faz log de itens refinados >= refine_items_log.
refine_items_log: 5
// Faz log de itens com taxa mínima de queda se for maior ou igual ao valor da função "rare_items_log".
// 1 = 0.01%, 100 = 1% chance de queda, etc
rare_items_log: 100
// Não faz log se o preço atual do item for menor que valor da função "price_items_log".
price_items_log: 1000
// Não faz log se a quantidade atual do item for menor que valor da função "amount_items_log".
amount_items_log: 100
//=============================================================

// Faz log do uso de galho seco? ( Nota 1)
log_branch: no

// Faz log de alterações de zeny?
// Configurações dos filtros
// 0 - não faz log; 1 - faz log de qualquer mudança; 2.....1000000 - quantidade minima de zenys para logar.
log_zeny: 1

// Faz log dos MvP mortos? ( Nota 1)
// Obsoleto. Use picklog ao invés deste. Pode ser útil para saber quantos MVPs são caçados.
log_mvpdrop: no

// Log de comandos ( Nota 1)
// Apenas comandos onde o grupos de jogadores ('conf/groups.conf') com
// 'log_commands' configurada para 'yes' serão gravadas.
log_commands: yes

// Registrar o comando de NPC 'logmes' (Nota 1)
log_npc: yes

// Registro de Chat (Global, Whisper, Grupo, Clã, Chat principal)
// FILTRO DE REGISTRO
// =============================================================
// 0 = Não registrar nada
// 1 - Mensagens globais
// 2 - Mensagens privadas (whisper)
// 4 - Mensagens de grupo
// 8 - Mensagens de Clã
// 16 - Mensagens do chat principal
// Exemplo:
// log_chat: 5 = registra mensagens privadas e de grupo
// log_chat: 8 = registra apenas mensagens de Clã
// log_chat: 31 = registra tudo
log_chat: 0

// Desabilitar registro de chat durante a WoE? (Nota 1)
log_chat_woe_disable: no

log_gm_db: atcommandlog
