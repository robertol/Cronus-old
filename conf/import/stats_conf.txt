// Nota 3: O valor é em bit. Caso não haja descrição,
//         assuma como valores (1: Pc, 2: Mob, 4: Pet, 8: Homun)

// O carregamento de habilidade deve ser cancelado quando inflingida por maldição/artordoar/sono/etc (incluí silêncio) (Nota 3)?
status_cast_cancel: 0

// Certas mudanças de status por habilidades devem ser removidas quando o jogador desconecta-se?
// Isto imita os servidores oficiais, onde não há recuperação de SP após usar Punho Extremo,
// Remover Equipamento, e outros buffs que são removidos na desconexão. A configuração é:
// 0 = não remover nada.
// 1 = remover buffs negativos (remoções, EF)
// 2 = remover buffs positivos (amplificar poder, corpo fechado...)
// 3 = remover ambos os buffs, positivos e negativos.
debuff_on_logout: 3

// Ajuste para a taxa natural de resistência para mudanças de status.
// Se 50, o status de defesa é reduzida pela metade, e você precisa de duas vezes mais status 
// para ignorá-las (eg: 200 vit para completamente ignorar atordoamento)
pc_status_def_rate: 330
mob_status_def_rate: 100

// Sorte (sor/luk) requerida para conseguir imunidade na mudança de status.
// Sorte aumenta a resistência completando o intervalo entre a resistência natural e a máxima 
// linearmente. Esta configuração indica a sorte requerida para ganhar imunidade completa.
// Eg: 40 vit -> 40% resist. 150 luk -> +50% de "intervalo completado". 
//     So 40% + (50% of 60%) = 70%
pc_luk_status_def: 330
mob_luk_status_def: 300

// Resistências máxima às mudanças de status. (100 = 100%)
// NOTA: Cartas e equipamentos podem ultrapassar esse limite, então isto só se aplica a resistência natural.
pc_max_status_def: 100
mob_max_status_def: 100
